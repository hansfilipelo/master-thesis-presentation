    \begin{overprint}

    \end{overprint}

\end{frame}

\begin{frame}

    \frametitle{Why does QUIC exist?}

    \framesubtitle{QUIC is a biproduct of the revolution HTTP/2}

    \begin{overprint}
        \uncover<+->{}
        \uncover<+->{
        Sending multiple resources over the same TCP connection makes HTTP/2 perform both better AND worse than HTTP/1.1.}
        ~\\
        ~\\
        \uncover<+->{A lot of information on the web urges people to switch to HTTP/2.}
    \end{overprint}

\end{frame}

% -------------------------------------------

\begin{frame}
    \frametitle{HTTP/2 on stable connections.}
    \begin{overprint}
        %\framesubtitle{HTTP/2 on stable connections.}

        \uncover<+->{}
        \uncover<+->{
        On stable connections HTTP/2 performs objectively better when fetching many resources. In \href{https://http2.akamai.com/demo}{some edge cases} \textbf{a lot} better.}
        ~\\
        ~\\
        \uncover<+->{
        In most web browsing however the user can't notice the difference because of processing delay from their browser.}

        \note{According to Bocchi et al. This is a combination of actual processing delay limitations and that many web pages are optimized to work well with HTTP/1.1}


    \end{overprint}

\end{frame}

% -------------------------------------------
\begin{frame}
    \frametitle{What about unstable connections then?}
    %\framesubtitle{}

        \only<2>{
        Sending multiple resources over one TCP connection becomes a problem. A packet-loss or out-of-order packet for one resource stream causes head-of-line blocking.}

\end{frame}

% ----------------------------------
\begin{frame}
    \frametitle{How is QUIC different?}
    \framesubtitle{QUIC improvements over HTTP/2}

    \begin{overprint}
        \uncover<+->{}

        \begin{itemize}
            \uncover<+->{
            \item QUIC includes per-stream congestion control, avoiding head-of-line blocking.}
            \uncover<+->{
            \item QUIC has way faster connection setup times (0-1 RTT vs 3 RTT).}
            \uncover<+->{
            \item Connection migration. Allows connection to continue even if user changes IP-address (moving between LTE/WiFi for example).}
            \uncover<+->{
            \item Forward error correction.}
        \end{itemize}

    \end{overprint}

\end{frame}
